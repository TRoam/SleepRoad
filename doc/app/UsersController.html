def punch
        @user = User.find(params[:id])
        @record = @user.records.last
        if @record and !@record.to 
            @record.to = Time.new()
            @record.during = Time.new() - @record.from
        elsif 
                @record = Record.new(
              from: Time.new(),
              user_id: params[:id]
              )
        end

        @record.save()
        render json: @record
end
